  <script>
    // Role config
    const roles = {
      admin: {
        icon: '🛡️',
        label: 'Admin',
        desc: 'Sign in with your administrator credentials',
        badgeClass: 'admin',
        btnText: 'Sign In as Admin',
      },
      management: {
        icon: '👨‍🏫',
        label: 'Management',
        desc: 'Sign in as HOD or Faculty',
        badgeClass: 'management',
        btnText: 'Sign In as Management',
      },
      student: {
        icon: '👤',
        label: 'Student',
        desc: 'Sign in to view your attendance records',
        badgeClass: 'student',
        btnText: 'Sign In as Student',
      },
    };

    // Tracks current active top-level tab ('admin'|'management'|'student')
    let _currentTab = 'admin';

    function selectRole(role, el) {
      _currentTab = role;

      // Update tabs
      document.querySelectorAll('.role-tab').forEach(t => t.classList.remove('active'));
      el.classList.add('active');

      const cfg = roles[role];

      // Update badge
      const badge = document.getElementById('role-badge');
      badge.textContent = cfg.icon + ' ' + cfg.label;
      badge.className = 'greeting-role ' + cfg.badgeClass;

      // Update desc
      document.getElementById('role-desc').textContent = cfg.desc;

      // Update button
      const btn = document.getElementById('login-btn');
      btn.textContent = cfg.btnText;
      btn.className = 'login-btn ' + cfg.badgeClass;

      // For admin/student, set role-input directly.
      // For management, keep role-input as whatever designation is active (faculty/hod).
      if (role === 'admin') {
        document.getElementById('role-input').value = 'admin';
      } else if (role === 'student') {
        document.getElementById('role-input').value = 'student';
      } else {
        // management tab — role-input should reflect the currently selected designation
        // read it from the segmented button that is active
        const activeDesig = document.querySelector('#row-designation .segmented button.active');
        document.getElementById('role-input').value = activeDesig ? activeDesig.dataset.role : 'faculty';
      }

      // Toggle form fields based on role
      const rowIdentity = document.getElementById('row-identity');
      const rowRoll     = document.getElementById('row-rollno');
      const rowDesig    = document.getElementById('row-designation');
      const rowEmp      = document.getElementById('row-empid');
      const rowPw       = document.getElementById('row-password');

      const usernameField = document.getElementById('username-field');
      const rollField     = document.getElementById('rollno-field');
      const empField      = document.getElementById('empid-field');
      const pwField       = document.getElementById('pw-field');

      // defaults — reset all
      rowIdentity.style.display = 'block';
      rowRoll.style.display     = 'none';
      rowDesig.style.display    = 'none';
      rowEmp.style.display      = 'none';
      rowPw.style.display       = 'block';

      usernameField.required = true;
      rollField.required     = false;
      empField.required      = false;
      pwField.required       = true;

      document.getElementById('identity-label').textContent = 'Username';
      document.getElementById('identity-icon').textContent  = '👤';

      if (role === 'student') {
        // Student: only roll number — no password needed on the UI
        // (view does a roll-number-only lookup with fallback password auth)
        rowIdentity.style.display = 'none';
        rowRoll.style.display     = 'block';
        rowPw.style.display       = 'none';
        usernameField.required    = false;
        pwField.required          = false;
        rollField.required        = true;
        rollField.focus();
      } else if (role === 'management') {
        rowDesig.style.display = 'block';
        rowEmp.style.display   = 'block';
        rowIdentity.style.display = 'none';  // hide username; employee_id is enough
        usernameField.required = false;
        empField.required      = true;
        empField.focus();
      } else {
        // admin
        usernameField.focus();
      }
    }

    // Called by Faculty / HOD segmented buttons
    function setDesignation(val, btnEl) {
      // Update the role hidden input — THIS is the fix for Faculty not working
      document.getElementById('role-input').value = val;

      const wrap = btnEl.parentElement;
      wrap.querySelectorAll('button').forEach(b => b.classList.remove('active'));
      btnEl.classList.add('active');

      // Update button label to match
      const btn = document.getElementById('login-btn');
      btn.textContent = val === 'hod' ? 'Sign In as HOD' : 'Sign In as Faculty';
    }

    // Password toggle
    function togglePw() {
      const f = document.getElementById('pw-field');
      const b = document.getElementById('pw-btn');
      if (f.type === 'password') {
        f.type = 'text';
        b.textContent = '🙈';
      } else {
        f.type = 'password';
        b.textContent = '👁️';
      }
    }

    // Add loading state on submit
    document.getElementById('login-form').addEventListener('submit', function() {
      const btn = document.getElementById('login-btn');
      btn.textContent = 'Signing in...';
      btn.style.opacity = '0.75';
      btn.style.pointerEvents = 'none';
    });
  </script>
